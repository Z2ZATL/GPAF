import tkinter as tk
import requests
from tkinter import messagebox

def get_weather():
    try:
        # ใช้ Open-Meteo API (ฟรี ไม่ต้องใช้ API key)
        url = 'https://api.open-meteo.com/v1/forecast?latitude=16.8211&longitude=100.2659&current_weather=true'
        r = requests.get(url, timeout=5)
        data = r.json()
        temp_c = data['current_weather']['temperature']
        temp_f = (temp_c * 9/5) + 32
        label_c['text'] = f'อุณหภูมิ: {temp_c:.1f} °C'
        label_f['text'] = f'Temperature: {temp_f:.1f} °F'
    except Exception as e:
        messagebox.showerror('Error', f'ไม่สามารถดึงข้อมูลอากาศ: {e}')

root = tk.Tk()
root.title('สภาพอากาศปัจจุบัน - เมืองพิษณุโลก')
root.geometry('300x150')

label_c = tk.Label(root, text='อุณหภูมิ: ... °C', font=('Tahoma', 14))
label_c.pack(pady=10)
label_f = tk.Label(root, text='Temperature: ... °F', font=('Tahoma', 14))
label_f.pack(pady=5)

btn = tk.Button(root, text='อัปเดตสภาพอากาศ', command=get_weather)
btn.pack(pady=10)

get_weather()
root.mainloop()
